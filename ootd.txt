        let currentLang = 'en';

        // Function to translate the page content
        const setLanguage = (lang) => {
            currentLang = lang;
            document.documentElement.lang = lang;
            const elements = document.querySelectorAll('[data-key]');
            
            elements.forEach(el => {
                const key = el.getAttribute('data-key');
                if (languageData[lang][key]) {
                    el.innerText = languageData[lang][key];
                }
            });

            // Handle special cases (placeholders) - Removed old token input logic

            // Update disclaimer below token button
            const disclaimerElement = document.querySelector('.text-xs.text-red-500.pt-2.text-center');
            if (disclaimerElement) {
                disclaimerElement.innerText = languageData[lang]['token_real_disclaimer'];
            }
        };

        // Initialize and Event Listeners
        document.addEventListener('DOMContentLoaded', () => {
            // Set initial language to English
            setLanguage(currentLang);

            // Smooth scroll for navigation
            document.querySelectorAll('a[href^="#"]').forEach(anchor => {
                anchor.addEventListener('click', function (e) {
                    e.preventDefault();
                    document.querySelector(this.getAttribute('href')).scrollIntoView({
                        behavior: 'smooth'
                    });
                });
            });

            // Language switcher event
            document.getElementById('lang-switcher-en').addEventListener('click', () => setLanguage('en'));
            document.getElementById('lang-switcher-zh').addEventListener('click', () => setLanguage('zh'));
        });
